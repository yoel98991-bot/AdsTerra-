// ========== REGISTRO DE NOTIFICACIONES PUSH PARA EL CLIENTE ========== async function registrarPushCliente(clienteId) { const Push = Capacitor.Plugins.PushNotifications; if (!Push) { console.warn('Plugin Push no disponible'); return; } let permiso = await Push.checkPermissions(); if (permiso.receive !== 'granted') { permiso = await Push.requestPermissions(); } if (permiso.receive !== 'granted') { console.warn('Permiso de notificaciones denegado'); return; } await Push.register(); Push.addListener('registration', async (token) => { console.log('Token FCM del cliente:', token.value); const { error } = await db.from('dispositivos').upsert({ user_id: clienteId, fcm_token: token.value, plataforma: 'android', updated_at: new Date().toISOString() }, { onConflict: 'user_id' }); if (!error) console.log('Token del cliente guardado en Supabase'); }); Push.addListener('pushNotificationReceived', (notif) => { console.log('Notificación recibida:', notif); // Opcional: mostrar un toast o actualizar el estado del pedido }); }
